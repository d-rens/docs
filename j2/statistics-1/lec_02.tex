\subsection{Conditional probability}
The probability of an event occurring given that some other event has already occured.

\[ P(A|B) = \text{The probability of event ``B'' occurring \textbf{given that} ``A'' has already occurred} .\] 

\subsubsection{Independent events} The occurrance of one event does
\textbf{not} affect the occurrance of another event. (Non-independent events
are Dependend)

\tbox{If A\&B are independent, \[ P(B|A)=P(B) .\] }

\ex{
\begin{enumerate}
    \item Roll a die: \textbf{Independent} 
        \[ P(2|3)= P(2) = \frac{1}{6} .\] 

    \item Drawing cards: 
        \[\text{with replacment:}\quad  P(Q|9) \frac{4}{52} = \frac{1}{13} \approx 0.077 .\] 
        \[\text{without replacment:}\quad  P(Q|9) = \frac{4}{51} \approx 0.078 .\] 

    \item Another card one:
        \[ \text{without replacment:} \quad P(Q|Q) = \frac{3}{51} \approx 0.059 .\] 
        \[ \text{with replacment:} \quad P(Q|Q) \frac{4}{52} \approx 0.77.\] 

    \item with replacment:
        \[ P(\heartsuit|J\diamondsuit)=\frac{13}{52} .\] 

    \item without
        \[ P(\heartsuit|J\diamondsuit)=\frac{13}{51}.\] 
\end{enumerate}
}


\subsection{Multiplication Rule}

\[ P(A \ and \ B)= P(A)\cdot P(B|A) .\] 
Recall: if independent $P(B|A)=P(B)$ so $P(A \ and \ B) = P(A)\cdot P(B)$.

\ex{
\textbf{Bag of marbles}

in the bag are 9 marbles:
\begin{itemize}
    \item 3 red
    \item 2 blue
    \item 4 green
\end{itemize}

with replacement:
\[
P(G \ and \ B) = P(G)\cdot P(B|G) =  \frac{4}{9}\cdot \frac{2}{9}= \frac{8}{81}
.\] 
 

without replacement:
\[
P(G \ and \ B) =P(G)\cdot P(B|G) = \frac{4}{9} \cdot \frac{2}{8} = \frac{1}{9}
.\] 


without replacement: \[
P(R \ and \ R)= \frac{3}{9}\cdot \frac{2}{8} = \frac{1}{12}
.\] 


without replacement: \[
P(B \ and \ B \ and \ B)= \frac{4}{9}\cdot \frac{3}{8}\cdot \frac{2}{7}= \frac{1}{21}
.\] 
}
