\section{nochmal integral einfuehrung}

\paragraph{Aufgabe 16}
a)
\begin{align*}
    \int_{0}^{t} 2 \ dx &= [2x]^t_0 \\ 
    2\cdot t-0&= 2t \\
\end{align*}

b)
\begin{align*}
    \int_{0}^{1} -x \ dx &= \left[  -\frac{x^2}{2}\right]_0^t   \\ 
    &= -\frac{t^2}{2} \\
\end{align*}

c)
\begin{align*}
    \int_{0}^{t} \frac{3}{2}x \ dx &= \left[  \frac{\frac{3}{2}x^2}{2}\right]_0^t  \\ 
        &= \frac{\frac{3}{2}t^2}{2} \\
\end{align*}
